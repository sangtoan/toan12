\section{Cực trị của hàm số} \subsection{Tóm tắt lý thuyết} \begin{dn} Hàm số $f\left( x \right)$ xác định trên $\mathscr{D}\subseteq \mathbb{R}$. \begin{itemize} \item $\checkmark$ Điểm ${x_0}\in \mathscr{D}$ được gọi là điểm cực đại của hàm số $f\left( x \right)$ nếu tồn tại một khoảng $\left( a;b \right)\subset \mathscr{D}$ sao cho ${x_0}\in \left( a;b \right)$ và $f\left( {x_0} \right)>f\left( x \right)$, $\forall x\in \left( a,b \right)\setminus \left\{ {x_0} \right\}$. \item $\checkmark$ Điểm $x_1\in \mathscr{D}$ được gọi là điểm cực tiểu của hàm số $f\left( x \right)$ nếu tồn tại một khoảng $\left( a;b \right)\subset \mathscr{D}$ sao cho $x_1\in \left( a;b \right)$ và $f\left( x_1 \right) < f\left( x \right)$, $\forall x\in \left( a,b \right)\setminus \left\{ {x_0} \right\}$. \end{itemize} \end{dn} \begin{luuy} Nếu hàm số $f(x)$ đạt cực đại (cực tiểu) tại $x_0$ thì \begin{itemize} \item $\checkmark$ $x_0$ được gọi là điểm cực đại (điểm cực tiểu) của hàm số. \item $\checkmark$ $f\left( x_0\right) $ được gọi là giá trị cực đại (giá trị cực tiểu) của hàm số, ký hiệu $f_{\text{CĐ}}$ $\left(f_{\text{CT}}\right) $. \item $\checkmark$ Điểm $M\left( x_0; f\left( x_0\right) \right) $ được gọi là điểm cực đại (điểm cực tiểu) của đồ thị hàm số. \end{itemize} \end{luuy} \subsubsection{Điều kiện cần và đủ để hàm số có cực trị} \begin{dl} \textbf{Điều kiện cần} Nếu hàm số $f\left( x \right)$ đạt cực trị tại điểm ${x_0}$ và hàm số có đạo hàm tại ${x_0}$, thì $f'\left( {x_0} \right)=0$. \end{dl} \begin{luuy} Điều ngược lại không đúng, nghĩa là $f'\left(x_0\right)=0$ thì $x=x_0$ chưa chắc đã là điểm cực trị của hàm số $y=f(x)$. \end{luuy} \begin{dl} \text{Điều kiện đủ} \begin{itemize} \item $\checkmark$ Nếu $f'\left( x \right) < 0,\forall x\in \left( a,{x_0} \right)$ và $f'\left( x \right)>0,\ \forall x\in \left( {x_0};b \right)$ thì $f\left( x \right)$ đạt cực tiểu tại ${x_0}$. \item $\checkmark$ Nếu $f'\left( x \right)>0,\forall x\in \left( a,{x_0} \right)$ và $f'\left( x \right) < 0,\ \forall x\in \left( {x_0};b \right)$ thì $f\left( x \right)$ đạt cực đại tại ${x_0}$. \end{itemize} \end{dl} Tức là, nếu đạo hàm của hàm số $y=f(x)$ đổi dấu từ âm sang dương khi qua ${x_0}$ Ta nói, đồ thị hàm số có điểm cực tiểu là $M\left( {x_0},{{y}_{\text{CT}}} \right)$. Nếu đạo hàm của hàm số $y=f\left(x\right)$ đổi dấu từ dương sang âm khi qua $x_0$. Ta nói đồ thị hàm số có điểm cực đại là $M\left( {x_0};{y_{\text{CĐ}}} \right)$. \begin{note} Không cần xét hàm số $y=f\left( x \right)$ có hay không đạo hàm tại ${x_0}$. \end{note} \begin{dl} Hàm số $y=f(x)$ có đạo hàm cấp một trên $(a;b)$ chứa $x_0$ mà $f'(x_0)=0$ và $y=f(x)$ có đạo hàm cấp hai khác không tại $x_0$. Khi đó, \begin{itemize} \item $\checkmark$ Nếu $f''\left( x_0 \right) > 0$ thì hàm số $y=f\left( x \right)$ đạt cực tiểu tại $x_0$. \item $\checkmark$ Nếu $f''\left( x_0 \right) < 0$ thì hàm số $y=f\left( x \right)$ đạt cực đại tại $x_0$. \end{itemize} \end{dl} \subsection{Các dạng toán và phương pháp giải} \begin{dang}{Tìm cực trị của hàm số $y=f(x)$ theo quy tắc } \textbf{Quy tắc I:} \begin{itemize} \item $\checkmark$ Tìm $f'(x).$ \item $\checkmark$ Tìm các điểm $x_i \,(i=1,2,\ldots,n )$ mà tại đó $f'(x_i)=0$ hoặc tại đó hàm số $f$ liên tục nhưng không có đạo hàm. \item $\checkmark$ Lập bảng biến thiên và kết luận về cực trị. \end{itemize} \textbf{Quy tắc II:} \begin{itemize} \item $\checkmark$ Tính $f'(x)$. \item $\checkmark$ Giải phương trình $f'(x)=0$ và tìm các nghiệm $x_i (i=1,2, \ldots, n).$ \item $\checkmark$ Tính $f''(x)$ và $f''(x_i)\, (i=1,2,\ldots, n)$ \begin{itemize} \item $\star$ $f''(x_i) < 0 \Rightarrow $ hàm số đạt cực đại tại $x_i.$ \item $\star$ $f''(x_i) > 0\Rightarrow $ hàm số đạt cực tiểu tại $x_i.$ \end{itemize} \end{itemize} \end{dang} \subsubsection{Ví dụ minh họa} \begin{vd}%[2D1B2] Tìm cực trị của hàm số $ y=x^3-3x^2-9x+5. $ \loigiai{ $\bullet$ Khi $x$ qua $-1$ thì $y'$ đổi dấu từ dương dang âm nên $-1$ là điểm cực đại của hàm số.\\ $\bullet$ Khi $x$ qua $3$ thì $y'$ đổi dấu từ âm dang dương nên $3$ là điểm cực tiểu của hàm số.\\ Tập xác định: $ \mathscr{D}=\mathbb{R}. $\\Ta có: $ y'=3x^2-6x-9$;
